### Github pages

```bash
npm install gh-pages --save-dev
```

```json
# package.json
{
  "scripts": {
    "deploy": "gatsby build --prefix-paths && gh-pages -d public",
  }
}
```

```js
// gatsby-config.js
module.exports = {
    pathPrefix: '/reading-group'
}
```
